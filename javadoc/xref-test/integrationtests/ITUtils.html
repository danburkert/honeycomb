95       * Inserts the number of specified rows with each row containing a column,
96       * {@link TestConstants#COLUMN1}, with the value provided and another column,
97       * {@link TestConstants#COLUMN2}, with an arbitrary value
98       *
99       * @param proxy          The proxy to use for invoking a scan, not null
100      * @param rowCount       The valid number of rows expected to be returned from the scan
101      * @param keyColumnValue The value stored in the first column, which is intended to be indexed
102      * @param uuid           The unique identifier to associate with each row inserted, not null
103      */
104     public static void insertData(final HandlerProxy proxy, final int rowCount,
105                                   final long keyColumnValue, final UUID uuid) {
106         checkNotNull(proxy);
107         verifyRowCount(rowCount);
108         checkNotNull(uuid);
109 
110         final Map<String, ByteBuffer> map = Maps.newHashMap();
111         map.put(TestConstants.COLUMN1, encodeValue(keyColumnValue));
112 
113         for (int x = 0; x < rowCount; x++) {
114             map.put(TestConstants.COLUMN2, encodeValue(x));
115             final Row row = new Row(map, uuid);
116             proxy.insertRow(row.serialize());
117         }
118 
119         proxy.flush();
120     }

180     private static void assertDifferentRows(final HandlerProxy proxy, final int rowCount) {
181         byte[] previous = null;
182 
183         for (int x = 0; x < rowCount; x++) {
184             final byte[] current = proxy.getNextRow();
185             assertNotNull(current);
186             assertThat(current, not(equalTo(previous)));
187             previous = current;

197     /**
198      * Check that the table open on the proxy has the expected number of data rows
199      * and index rows on each index (checks both ascending and descending
200      * directions).  Note: this could be very slow for big tables.
201      * @param proxy HandlerProxy with table already open
202      * @param schema TableSchema of open table
203      * @param expectedRowCount Expected number of rows
204      */
205     public static void assertRowCount(final HandlerProxy proxy,
206                                       final TableSchema schema,
207                                       final long expectedRowCount) {
208         checkState(proxy.getTableName() != null, "Proxy must have an open table.");
209         checkNotNull(schema);
210         verifyRowCount(expectedRowCount);
211 
212         proxy.startTableScan();
213         assertResultCount(proxy, expectedRowCount);
214         proxy.endScan();
215 
216         QueryKey queryKey;
217         for(IndexSchema indexSchema : schema.getIndices()) {
218             queryKey = new QueryKey(indexSchema.getIndexName(),
219                     QueryType.INDEX_FIRST, ImmutableMap.<String, ByteBuffer>of());
220             proxy.startIndexScan(queryKey.serialize());
221             assertResultCount(proxy, expectedRowCount);
222             proxy.endScan();
223 
224             queryKey = new QueryKey(indexSchema.getIndexName(),
225                     QueryType.INDEX_LAST, ImmutableMap.<String, ByteBuffer>of());
226             proxy.startIndexScan(queryKey.serialize());
227             assertResultCount(proxy, expectedRowCount);
228             proxy.endScan();
229         }
230     }

246     /**
247      * Start concurrency number of actions concurrently and at the same time.  This
248      * method ensures that the actions are started at the same time, and they are
249      * not being blocked in a thread pool.  Obviously, this uses concurrency
250      * number of threads simultaneously so use caution as thread starvation
251      * deadlock can occur with high concurrency levels.
252      *
253      * @param concurrency Number of handler proxies to run the action against, concurrently
254      * @param setup Action to be run on each proxy before starting concurrent actions
255      * @param action Action to be run concurrently and in sync with other proxies
256      * @param cleanup Action to be run to cleanup proxy
257      * @param factory HandlerProxyFactory to supply proxies to be run concurrently
258      * @throws InterruptedException
259      */
260     public static void startProxyActionConcurrently(int concurrency,
261                                                     final ProxyRunnable setup,
262                                                     final ProxyRunnable action,
263                                                     final ProxyRunnable cleanup,
264                                                     final HandlerProxyFactory factory)
265             throws InterruptedException{
266         final ExecutorService executor = Executors.newCachedThreadPool();
267         final CountDownLatch ready = new CountDownLatch(concurrency);
268         final CountDownLatch start = new CountDownLatch(1);
269         final CountDownLatch done = new CountDownLatch(concurrency);
270 
271         for (int i = 0; i < concurrency; i++) {
272             executor.execute(new Runnable() {
273                 @Override
274                 public void run() {
275                     HandlerProxy proxy = factory.createHandlerProxy();
276                     setup.run(proxy);
277                     ready.countDown();
278                     try {
279                         start.await();
280                         action.run(proxy);
281                     } catch (InterruptedException e) {
282                         System.out.println("Interuppted");
283                         Thread.currentThread().interrupt();
284                     }
285                     cleanup.run(proxy);
286                     done.countDown();
287                 }
288             });
289         }
290         ready.await();
291         start.countDown();
292         done.await();
293     }
