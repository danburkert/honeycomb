25  import java.nio.ByteBuffer;
26  import java.util.ArrayList;
27  import java.util.Arrays;
28  import java.util.List;

30  import static com.google.common.base.Preconditions.checkArgument;
31  import static com.google.common.base.Preconditions.checkNotNull;
32  
33  /**
34   * Helper functions for variable length encoding data in a binary-sort safe
35   * manner.
36   */
37  public class VarEncoder {
38      /**
39       * Variable length encodes a long.
40       *
41       * @param value Long value
42       * @return Variable length encoded bytes
43       */
44      public static byte[] encodeULong(long value) {
45          checkArgument(value >= 0, "Encoded long must be non-negative");
46          int size = varLongSize(value);
47          byte[] encodedValue = new byte[1 + size];
48          encodedValue[0] = (byte) size;
49          for (int i = size; i > 0; i--) {
50              encodedValue[i] = (byte) value;
51              value >>>= 8;
52          }
53          return encodedValue;
54      }
55  
56      /**
57       * Decode a variable length encoded long in a byte array
58       *
59       * @param encodedValue Encoded byte string
60       * @return Long value
61       */
62      public static long decodeULong(final byte[] encodedValue) {
63          checkNotNull(encodedValue);
64          long value = 0;
65          long mask;
66          for (int i = 1; i < encodedValue.length; i++) {
67              mask = 0xFF & encodedValue[i];
68              value = value << 8;
69              value |= mask;
70          }
71          return value;
72      }
73  
74      /**
75       * Combine many byte arrays into one
76       *
77       * @param arrays List of byte arrays
78       * @return Combined byte array
79       */
80      public static byte[] appendByteArrays(List<byte[]> arrays) {
81          checkNotNull(arrays);
82          int size = 0;
83          for (byte[] array : arrays) {
84              size += array.length;
85          }
86          ByteBuffer bb = ByteBuffer.allocate(size);
87          for (byte[] array : arrays) {
88              bb.put(array);
89          }
90          return bb.array();
91      }
92  
93      /**
94       * Combine many byte arrays into one with a prefix at the beginning of the combined array.
95       *
96       * @param prefix Byte prefix
97       * @param arrays Many byte arrays
98       * @return Combined byte array
99       */
100     public static byte[] appendByteArraysWithPrefix(byte prefix, byte[]... arrays) {
101         checkNotNull(prefix);
102         List<byte[]> elements = new ArrayList<byte[]>();
103         byte[] prefixBytes = {prefix};
104         elements.add(prefixBytes);
105         elements.addAll(Arrays.asList(arrays));
106         return appendByteArrays(elements);
107     }
108 
109     private static int varLongSize(final long value) {
110         if ((value & (0xffffffffffffffffL << 8)) == 0) return 1;
111         if ((value & (0xffffffffffffffffL << 16)) == 0) return 2;
112         if ((value & (0xffffffffffffffffL << 24)) == 0) return 3;
113         if ((value & (0xffffffffffffffffL << 32)) == 0) return 4;
114         if ((value & (0xffffffffffffffffL << 40)) == 0) return 5;
115         if ((value & (0xffffffffffffffffL << 48)) == 0) return 6;
116         if ((value & (0xffffffffffffffffL << 56)) == 0) return 7;
117         return 8;
118     }
